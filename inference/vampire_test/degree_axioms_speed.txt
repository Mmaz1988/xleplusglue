tff(fast_type, type, fast: ($i * $int) > $o).
tff(slow_type, type, slow: ($i * $int) > $o).
tff(kind_type, type, kind: ($i * $i) > $o).

tff(adj_postulate, axiom,
![X: $i, Y: $i , D: $int]: (arg1(X, Y) & fast(X, D)  <=> fast(Y, D))).

%FAST
tff(downward_monotonicity_fast, axiom,
    ![X: $i, Delta1: $int]:
    (fast(X, Delta1) <=>
         ![Delta2: $int]:
             ($lesseq(Delta2,Delta1) => fast(X, Delta2))
        )
).

tff(consistency_axiom, axiom,
    ![X: $i, Y: $i]:
        (? [D: $int]: (fast(X, D) & ~fast(Y, D))
         =>
         ![D2: $int]: (fast(Y, D2) => fast(X, D2)))
).

tff(existence_of_maximal_fast, axiom,
    ![X: $i]:(
        ?[Delta1: $int]:
             ( fast(X, Delta1) &
             ~?[Delta2: $int]:
                 ($greater(Delta2,Delta1) & fast(X, Delta2))
            )
    )
).
