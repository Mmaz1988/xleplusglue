%Declare predicate to identify fast
tff(fast_type, type, fast: ($i * $int) > $o).
tff(slow_type, type, slow: ($i * $int) > $o).
tff(kind_type, type, kind: ($i * $i) > $o).
%Does slow have a maximum? more like clean? 

%FAST
%possibly optional (taken from Haruta, et al. Github)
tff(downward_monotonicity_fast, axiom,
    ![X: $i, Delta1: $int]:
    (fast(X, Delta1) <=>
         ![Delta2: $int]:
             ($lesseq(Delta2,Delta1) => fast(X, Delta2))
        )
).

%consistency postulate, p. 148, Haruta, et al. 2022
tff(consistency_axiom, axiom,
    ![X: $i, Y: $i]:
        (? [D: $int]: (fast(X, D) & ~fast(Y, D))
         =>
         ![D2: $int]: (fast(Y, D2) => fast(X, D2)))
).

%supremum, p. 149, Haruta, et al. 2022
tff(existence_of_maximal_fast, axiom,
    ![X: $i]:(
        ?[Delta1: $int]:
             ( fast(X, Delta1) &
             ~?[Delta2: $int]:
                 ($greater(Delta2,Delta1) & fast(X, Delta2))
            )
    )
).

%ANTONYMY
%antonyms, p. ?, Haruta, et al. 2022
tff(antonymy_relation_1, axiom,
    ![X: $i, Delta1: $int]:(
	 slow(X,Delta1) =>
	    ![Delta2: $int]:(
	        $less(Delta1, Delta2) =>
			   ~fast(X,Delta2))
			  )
   ).

tff(antonymy_relation2, axiom,
    ![X: $i, Delta1: $int]:(
	 fast(X,Delta1) =>
	    ![Delta2: $int]:(
	    ![Delta2: $int]:(
	        $less(Delta2, Delta1) =>
			   ~slow(X,Delta2)))
   ).
