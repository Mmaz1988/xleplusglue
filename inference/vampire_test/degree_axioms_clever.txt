%Declare predicate to identify clever
tff(clever_type, type, height: ($i * $int) > $o).
tff(max_clever, type, max_height: ($i * $int) > $o).

%possibly optional (taken from Haruta, et al. Github)
tff(downward_monotonicity_fast, axiom,
    ![X: $i, Delta1: $int]:
    (fast(X, Delta1) <=>
         ![Delta2: $int]:
             ($lesseq(Delta2,Delta1) => fast(X, Delta2))
        )
).

%consistency postulate, p. 148, Haruta, et al. 2022
tff(consistency_axiom, axiom,
    ![X: $i, Y: $i]:
        (? [D: $int]: (fast(X, D) & ~fast(Y, D))
         =>
         ![D2: $int]: (fast(Y, D2) => fast(X, D2)))
).

%supremum, p. 149, Haruta, et al. 2022
tff(existence_of_maximal_fast, axiom,
    ![X: $i]:(
        ?[Delta1: $int]:
             ( fast(X, Delta1) &
             ~?[Delta2: $int]:
                 ($greater(Delta2,Delta1) & fast(X, Delta2))
            )
    )
).
